// app/posts/page.tsx
export async function generateStaticParams() {
  const posts = await fetch('https://api.example.com/posts').then(res => res.json());
  
  return posts.map(post => ({
    slug: post.slug,
  }));
}

最佳实践

在使用 Next.js 13 开发时，建议遵循以下最佳实践：

  • 合理使用服务端组件和客户端组件
  • 利用静态生成提升性能
  • 实施增量静态再生成（ISR）
  • 优化图片和字体加载
  • 合理规划路由结构

总结

Next.js 13 通过 App Router 和 React Server Components 等创新特性，为现代 Web 应用开发提供了更强大的工具和更好的性能。通过合理运用这些特性，我们可以构建出更快速、更可靠的应用。
